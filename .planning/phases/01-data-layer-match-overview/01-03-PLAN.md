```typescript
import { betterAuth } from "better-auth";
import { drizzleAdapter } from "better-auth/adapters/drizzle";
import { emailOTP } from "better-auth/plugins";
import { tanstackStartCookies } from "better-auth/tanstack-start";
import { db } from "./db";
import { adminOtp, sendEmail } from "./email";

const ALLOWED_EMAILS = (process.env.ADMIN_EMAILS || "")
  .split(",")
  .map((e) => e.trim().toLowerCase())
  .filter(Boolean);

export function isAllowedEmail(email: string): boolean {
  return ALLOWED_EMAILS.includes(email.toLowerCase().trim());
}

export const auth = betterAuth({
  database: drizzleAdapter(db, { provider: "pg" }),
  secret: process.env.BETTER_AUTH_SECRET,
  baseURL: process.env.BETTER_AUTH_URL || "http://localhost:3001",
  plugins: [
    emailOTP({
      async sendVerificationOTP({ email, otp, type }) {
        if (type !== "sign-in") return;
        // Defense in depth: don't send OTP to non-admin emails
        if (!ALLOWED_EMAILS.includes(email.toLowerCase())) {
          console.log(`[admin] OTP blocked for non-admin email: ${email}`);
          return;
        }
        console.log(`[admin] OTP for ${email}: ${otp}`);
        await sendEmail(email, adminOtp(otp));
      },
      otpLength: 6,
      expiresIn: 300,
    }),
    tanstackStartCookies(), // MUST be last plugin (Pitfall 8)
  ],
});
```

2. Rewrite `apps/admin/src/lib/auth-session.ts` to use Better Auth's session API instead of the old custom parseSessionToken/getSession:
```typescript
import { createServerFn } from "@tanstack/react-start";

export const getAuthSession = createServerFn().handler(async () => {
  const { getRequestHeaders } = await import("@tanstack/react-start/server");
  const { auth } = await import("~/lib/auth");
  const { isAllowedEmail } = await import("~/lib/auth");

  const headers = new Headers(getRequestHeaders());
  const session = await auth.api.getSession({ headers });

  if (!session?.user?.email) {
    return { email: null, isAuthenticated: false };
  }

  // Defense in depth: verify email is still in allowlist even after session creation
  if (!isAllowedEmail(session.user.email)) {
    return { email: null, isAuthenticated: false };
  }

  return { email: session.user.email, isAuthenticated: true };
});
```

Note: The import style (dynamic import inside handler) matches the existing pattern in the current auth-session.ts. Using `getRequestHeaders` (plural) may need verification — check the TanStack Start server API. If the API is `getRequestHeader` (singular, taking a header name), adapt accordingly. The key is to pass request headers to `auth.api.getSession()` so Better Auth can read the session cookie.
